document.addEventListener("DOMContentLoaded", function () { document.querySelectorAll('.section__header').forEach((containerEl) => { const $slider = $(containerEl.querySelector('.slider')); const prevBtn = containerEl.querySelector('.slider-prev'); const nextBtn = containerEl.querySelector('.slider-next'); $slider.slick({ slidesToShow: 3, //화면에 보여질 요소의 개수 slidesToScroll: 1, // 이동개수 arrows: false, //화살표 안보이기 dots: false, //점 안보이기 infinite: false // 슬라이드 가 처음/끝에 멈춤 }); // 각 외부 버튼으로 컨트롤 prevBtn.addEventListener('click', function(){ $slider.slick('slickPrev') }); nextBtn.addEventListener('click', () => { $slider.slick('slickNext') }); });
